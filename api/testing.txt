products = Product.objects.all()

    newest_prices = PriceHistory.objects.order_by(
        'product',
        'store',
        '-date'
    ).distinct('product', 'store')

    price_map = {}

    for price in newest_prices:
        price_id = price.product.product_id

        if price_id not in price_map:
            price_map[price_id] = []

        price_map[price_id].append({
            "store": price.store.name,
            "location": price.store.location,
            "price": float(price.price)
        })

    data = []

    for product in products:
        data.append({
            "id": product.product_id,
            "name": product.name,
            "unit": product.unit,
            "price": price_map.get(product.product_id, [])
        })

    return JsonResponse(data, safe = False)
